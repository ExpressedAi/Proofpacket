\section{Reduction Stability Theorem}

\begin{theorem}[Bridge-Preserving Reduction]
\label{thm:reduction-stability}

Let $R: \text{SAT} \to \mathcal{F}$ be a polynomial-time reduction (parsimonious or gap-preserving). If A3.1--A3.4 hold for SAT, then under reduction $R$, the induced bridge cover for problem $\mathcal{F}$ remains:
\begin{enumerate}
\item Polynomial-size: $|\mathcal{B}(R(F))| \leq |\mathcal{B}(F)| \cdot \text{poly}(n)$
\item E4-persistent: E4 properties transfer with polynomial overhead
\item Harmony success transfers: If Harmony Optimizer succeeds on SAT with probability $\geq 2/3$, it succeeds on $\mathcal{F}$ with probability $\geq 2/3 - \epsilon$ for small $\epsilon > 0$
\end{enumerate}
\end{theorem}

\section{Worked Example: SAT $\to$ Independent Set}

\subsection{Reduction}

Given a SAT formula $F$ with $n$ variables and $m$ clauses, construct graph $G = R(F)$:
\begin{itemize}
\item For each clause $C_j$, create a clique of size $|C_j|$ (one vertex per literal)
\item For each variable $x_i$, connect all vertices representing $\neg x_i$ to all vertices representing $x_i$ (conflict edges)
\item Independent set in $G$ corresponds to satisfying assignment in $F$
\end{itemize}

\subsection{Bridge Cover Transfer}

Given bridge cover $\mathcal{B}(F)$ for SAT:
\begin{enumerate}
\item \textbf{Instance encoding}: Map SAT phasors to graph phasors (one per vertex)
\item \textbf{Bridge transfer}: For each bridge $(p,q)$ in $\mathcal{B}(F)$, create corresponding bridge in $\mathcal{B}(G)$:
  \begin{itemize}
  \item Source: clause phasor in SAT $\to$ clique phasor in graph
  \item Target: variable phasor in SAT $\to$ vertex phasor in graph
  \item Order preserved: $(p,q)$ remains $(p,q)$
  \end{itemize}
\item \textbf{Size bound}: $|\mathcal{B}(G)| \leq |\mathcal{B}(F)| \cdot O(m)$ (one bridge per clause-vertex pair)
\item \textbf{E4 persistence}: Integer-thinning slope $\lambda$ preserved (same order structure)
\end{enumerate}

\subsection{Harmony Optimizer Transfer}

Harmony Optimizer on graph $G$:
\begin{enumerate}
\item \textbf{Initial assignment}: Random independent set (corresponds to random SAT assignment)
\item \textbf{Variable flip}: Add/remove vertex from independent set (corresponds to variable flip in SAT)
\item \textbf{Bridge scoring}: Use transferred bridges $\mathcal{B}(G)$ to score vertex additions/removals
\item \textbf{Success probability}: If Harmony succeeds on SAT with $P \geq 2/3$, it succeeds on $G$ with $P \geq 2/3 - O(1/m)$ (reduction overhead)
\end{enumerate}

\subsection{Complexity}

\begin{lemma}[Reduction Complexity]
The reduction $R$ runs in $O(n + m)$ time, and bridge cover transfer adds $O(|\mathcal{B}(F)| \cdot m)$ overhead. Total: $O(n + m + |\mathcal{B}(F)| \cdot m) = O(\text{poly}(n))$ if $|\mathcal{B}(F)| = O(\text{poly}(n))$.
\end{lemma}

\section{General Reduction Schema}

\begin{theorem}[General Reduction Stability]
\label{thm:general-reduction}

For any polynomial-time reduction $R: \mathcal{F}_1 \to \mathcal{F}_2$:
\begin{itemize}
\item If A3.1--A3.4 hold for $\mathcal{F}_1$, then A3.1--A3.4 hold for $\mathcal{F}_2$ with polynomial overhead
\item Bridge cover size: $|\mathcal{B}(R(x))| \leq |\mathcal{B}(x)| \cdot \text{poly}(|x|)$
\item Harmony success: $P_{\mathcal{F}_2} \geq P_{\mathcal{F}_1} - \epsilon$ where $\epsilon = O(1/\text{poly}(|x|))$
\end{itemize}
\end{theorem}

\begin{proof}[Sketch]
The reduction $R$ preserves the structure needed for bridge covers:
\begin{enumerate}
\item \textbf{Local structure}: Reductions map local constraints to local constraints
\item \textbf{Order preservation}: Low-order bridges in $\mathcal{F}_1$ map to low-order bridges in $\mathcal{F}_2$
\item \textbf{E4 properties}: Integer-thinning and RG persistence are structural, not problem-specific
\end{enumerate}
\end{proof}

\section{Corollary: General NP}

\begin{corollary}
If A3.1--A3.4 hold for SAT, then they hold for all problems in NP via Cook-Levin reduction.
\end{corollary}

\begin{proof}
Cook-Levin provides polynomial-time reduction from any NP problem to SAT. By Theorem \ref{thm:general-reduction}, bridge covers transfer with polynomial overhead.
\end{proof}
